#include <iostream>
using namespace std;
template<class T>
struct Node{
    Node*next;
    T data;
    Node(T val){
        data=val;
        next=NULL;
    }
};
template<class T>
class list{
  Node<T>*head;
  int count;
  public:
  list(){
      head=NULL;
      count=0;
      
  }
  bool empty(){
      return count==0;
  }
  int size(){
      return count;
  }
  void insert(T val,int pos){
      if(pos<0||pos>count)
      return;
      Node<T>*temp=new Node<T>(val);
      if(pos==0){
    temp->next=head;
    head=temp;
    count++;
    return;
}

      
      Node<T>*curr=head;
      for(int i=0;i<pos-1;i++){
          curr=curr->next;
      }
      temp->next=curr->next;
      curr->next=temp;
         count++;
      
  }
  void display(){
      if(empty())
      return;
      Node<T>*curr=head;
      for(int i=0;i<size();i++){
          cout<<curr->data<<"  ";
          curr=curr->next;
      }
      cout<<endl;
  }
 T retrive(int pos){
      if(pos<0||pos>=count)
      return 0;
      Node<T>*ar=head;
      for(int i=0;i<pos;i++){
          ar=ar->next;
      }
      return ar->data;
  }
bool remove(int pos){
    if(pos<0||pos>count)
    return 0;
    Node<T>*temp=head;
if(pos==0){
    Node<T>*temp=head;
    head=head->next;
    delete temp;
}
else{
for(int i=0;i<pos-1;i++){
    temp=temp->next;
}
Node<T>*curr=temp->next;
temp->next=curr->next;
delete curr;

}
count--;
return 1;
}



list(const list<T>&l1) {
    head = NULL;
    count = 0;

    Node<T>* temp =l1.head;

    while(temp != NULL){
        insert(temp->data,count);
        temp = temp->next;
    }
}
list<T>& operator=(const list<T>& l1){
   
    if(this == &l1)
        return *this;

    while(!empty())
        remove(0);

    Node<T>* temp = l1.head;

    while(temp != NULL){
        insert(temp->data,count);
        temp=temp->next;
    }

    return *this;
}
void duplicates(){
    Node<T>*temp=head;
if(temp==NULL)
return;
while(temp!=NULL&&temp->next!=NULL){
if(temp->data==temp->next->data){
Node<T>*sec=temp->next;
temp->next=sec->next;
delete sec;
count--;
}
else{
    temp=temp->next;
}
}
return;
}
void evenVal(){
   
    if(head==NULL)
    return;
    if(head->data%2==0){
        Node<T>*ar=head;
        head=head->next;
        delete ar;
        count--;
    }
    Node<T>*temp=head;
    while(temp!=NULL&&temp->next!=NULL){
        if(temp->next->data%2==0){
       Node<T>*sec=temp->next;
        temp->next=sec->next;
        delete sec;
count--;
        
        }
    
    else{
        temp=temp->next;
    }
    }
    return;
}
};
