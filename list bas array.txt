#include <iostream>

using namespace std;
const int maxSize= 100;


template <class T>
class List{
    T arr[maxSize];
    int count;

public:
    List(){
        count=0;
    }


    bool empty(){
        return count==0;
     }

     int size(){
         return count;
     }
    void insert(int pos, T val){
      if(maxSize == count || pos<0 || pos> count) return;

      for(int i= count-1; i>= pos; i--)
        arr[i+1]= arr[i];

      arr[pos]= val;
      count++;
    }


    void remove(int pos){
        if(empty() || pos<0 || pos> count) return;

        for(int i=pos; i< count; i++)
            arr[i] = arr[i+1];

       count--;

    }

    void replace(int pos, T newVal){
        if(empty() || pos<0 || pos> count) return;

        arr[pos] = newVal;
    }


    T retreive(int pos){
      if(empty() || pos<0 || pos> count) throw runtime_error("Invalid");

      return arr[pos];
    }
};
