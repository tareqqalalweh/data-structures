#include <iostream>
using namespace std;

const int n = 11;

template<class T>
struct node
{
    T data;
    node<T>* next;

    node()
    {
        next = 0;
    }

    node(T item)
    {
        data = item;
        next = 0;
    }
};


template <class T>
class hashh {

private:

    node<T>* table[n];
    int count;

    int hashfunc(T key)
    {
        return key % n;
    }

public:

    hashh()
    {
        count = 0;

        for (int i = 0; i < n; i++)
            table[i] = NULL;
    }


    bool add(T key)
    {
        int pos = hashfunc(key);

        node<T>* temp = new node<T>(key);

        temp->next = table[pos];
        table[pos] = temp;

        count++;

        return 1;
    }


    bool del(T key)
    {
        if (empty())
            return 0;

        int pos = hashfunc(key);

        if (table[pos] == 0)
            return 0;

        node<T>* prev;
        node<T>* follow;

        prev = follow = table[pos];


        while (follow != 0 && follow->data != key)
        {
            prev = follow;
            follow = follow->next;
        }


        if (follow == 0)
            return 0;


        if (prev == follow)
            table[pos] = follow->next;

        else
            prev->next = follow->next;


        delete follow;
        count--;

        return 1;
    }


    bool find(T key)
    {
        if (empty())
            return 0;

        int pos = hashfunc(key);

        node<T>* temp = table[pos];


        while (temp != 0 && temp->data != key)
        {
            temp = temp->next;
        }


        if (temp == 0)
            return 0;

        return 1;
    }


    void print()
    {
        for (int i = 0; i < n; i++)
        {
            node<T>* temp = table[i];

            cout << "Row " << i << " is : ";

            while (temp != 0)
            {
                cout << temp->data << "  ";
                temp = temp->next;
            }

            cout << endl;
        }
    }


    int size()
    {
        return count;
    }


    bool empty()
    {
        return count == 0;
    }

};


int main()
{
    hashh<int> h;


    // Add
    h.add(10);
    h.add(21);
    h.add(32);
    h.add(15);


    cout << "Hash Table:\n";
    h.print();


    cout << endl;


    // Find
    if (h.find(21))
        cout << "21 Found" << endl;
    else
        cout << "21 Not Found" << endl;



    // Delete
    cout << "\nDelete 21\n";

    h.del(21);


    h.print();


    cout << "\nSize = " << h.size();


    return 0;
}
