#include<iostream>
using namespace std;
template<class T>
struct Node{
    Node* next,*prev;
    T data;
Node(int val){
    data=val;
    next=prev=NULL;
}
};
template<class T>
class List{
    int count;
    Node<T>* head,*tail;

public:

    List(){
        count = 0;
        head = tail = NULL;
    }
    bool empty(){
        return count==0;
    }
     int size(){
         return count;
     }
    void insert(int pos, T val){
        if(pos<0||pos>count)
            return;
        Node<T>* temp = new Node<T>(val);
        if(head==NULL){
            head = tail = temp;
            count++;
        }
        else if(pos==0){
            temp->next = head;
            head->prev = temp;
            head = temp;
            count++;
        }
        else if(pos==count){
            temp->prev = tail;
            tail->next = temp;
            tail = temp;
            count++;
        }

        else{

            Node<T>* rer = head,*sec;

            for(int i=0;i<pos-1;i++)
            rer = rer->next;
             sec = rer->next;
            rer->next = temp;
            temp->prev = rer;
            temp->next = sec;
            sec->prev = temp;

            count++;
        }
    }
    void display(){
        Node<T>*temp=head;
        while(temp!=NULL){
            cout<<temp->data<<"  ";
            temp=temp->next;
        }
    }
    
    void remove(int pos){
        if(empty() || pos<0 || pos> count) return;
        if(pos==0){
            Node<T>*temp=head;
            head=head->next;
            head->prev=NULL;
            delete temp;
            return ;
        }
        
        Node<T>*tem=head,*sec;
        for(int i=0; i< pos; i++){
            sec=tem;
            tem=tem->next;
        }
           
          sec->next=tem->next;
           tem->prev=sec;
        

       count--;

    }
};
